# Profiler workflow for Dynamo 1.0.1 DGDRs
# The profiler runs only with disaggregated deployments (decode + prefill).
# The DGDR examples already embed the base disaggregated DGD under spec.overrides.dgd.
# Prerequisites: Dynamo 1.0.1 installed, hf-token-secret in namespace, GPU nodes

export NAMESPACE=dynamo-system

# 1) Submit a profiling request (pick one)
# Rapid search (AI Configurator-backed where supported):
kubectl apply -f profile_sla_aic_dgdr.yaml
# Thorough online sweep on real GPUs:
# kubectl apply -f profile_sla_online_dgdr.yaml

# 2) Watch DGDR state
kubectl get dgdr -n $NAMESPACE
kubectl describe dgdr qwen-sla-aic -n $NAMESPACE

# 3) Tail profiler job logs
kubectl logs -f job/profile-qwen-sla-aic -n $NAMESPACE

# 4) Inspect outputs
kubectl get configmap dgdr-output-qwen-sla-aic -n $NAMESPACE -o yaml

# 5) Extract the generated DGD from DGDR status
kubectl get dgdr qwen-sla-aic -n $NAMESPACE \
  -o jsonpath='{.status.profilingResults.selectedConfig}' | yq

# 6) Inspect the auto-applied deployment
kubectl get dynamographdeployment qwen-sla-aic -n $NAMESPACE -o yaml
