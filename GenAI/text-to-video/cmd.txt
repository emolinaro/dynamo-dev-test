# vLLM-Omni text-to-video with Wan-AI/Wan2.1-T2V-1.3B-Diffusers
kubectl apply -f vllm-agg_wan_t2v.yaml
kubectl -n dynamo-system get dynamographdeployment vllm-agg-wan-t2v
kubectl -n dynamo-system port-forward svc/vllm-agg-wan-t2v-frontend 8000:8000 &

curl -s http://127.0.0.1:8000/v1/videos \
  -H "Content-Type: application/json" \
  -d '{
    "model": "Wan-AI/Wan2.1-T2V-1.3B-Diffusers",
    "prompt": "A drone flyover of a mountain landscape",
    "seconds": 2,
    "size": "832x480",
    "response_format": "url"
  }' > response.json

cat response.json | jq .

# If the response contains a file:// URL, copy the MP4 out of the worker pod:
VIDEO_URL=$(jq -r '.data[0].url' response.json)
VIDEO_PATH=${VIDEO_URL#file://}
VIDEO_NAME=$(basename "$VIDEO_PATH")
WORKER_POD=$(kubectl -n dynamo-system get pods -o name | grep vllm-agg-wan-t2v-omnivideoworker | head -n 1 | sed 's#pod/##')

mkdir -p ~/Downloads/wan-t2v
kubectl -n dynamo-system cp "${WORKER_POD}:${VIDEO_PATH}" "$HOME/Downloads/wan-t2v/${VIDEO_NAME}"
