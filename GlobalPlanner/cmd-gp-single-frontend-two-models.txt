# Single-frontend, two-model vLLM global-planner example adapted from
# the upstream global-planner-vllm-test.yaml.
#
# Required:
#   export STORAGE_CLASS_NAME=<rwx-storage-class>
#   export DYNAMO_IMAGE=<frontend/router/planner image>
#   export DYNAMO_VLLM_IMAGE=<worker image>
#
# Note:
#   STORAGE_CLASS_NAME must support ReadWriteMany for the shared model-cache PVC.
#   Do not use csi-rbd-sc for this example on clusters where it is RWO-only.
#
# Example image values for this repo:
#   export DYNAMO_IMAGE=nvcr.io/nvidia/ai-dynamo/vllm-runtime:1.0.1
#   export DYNAMO_VLLM_IMAGE=nvcr.io/nvidia/ai-dynamo/vllm-runtime:1.0.1
#
# Example model IDs (1 GPU per prefill/decode worker on H100-class GPUs; override as needed):
#   export MODEL_A=meta-llama/Llama-3.2-3B-Instruct
#   export MODEL_B=Qwen/Qwen2.5-7B-Instruct

export K8S_NAMESPACE=${K8S_NAMESPACE:-dynamo-system}
export MODEL_A=${MODEL_A:-meta-llama/Llama-3.2-3B-Instruct}
export MODEL_B=${MODEL_B:-Qwen/Qwen2.5-7B-Instruct}
export PROFILE_RESULTS_STAGING_ROOT=${PROFILE_RESULTS_STAGING_ROOT:-/profile-results}
export MODEL_A_PROFILE_RESULTS_SOURCE_DIR=${MODEL_A_PROFILE_RESULTS_SOURCE_DIR:-/workspace/tests/planner/profiling_results/H200_TP1P_TP1D}
export MODEL_B_PROFILE_RESULTS_SOURCE_DIR=${MODEL_B_PROFILE_RESULTS_SOURCE_DIR:-/workspace/tests/planner/profiling_results/H200_TP1P_TP1D}
export MODEL_A_PROFILE_RESULTS_ALIAS_NAME=${MODEL_A_PROFILE_RESULTS_ALIAS_NAME:-H100_TP1P_TP1D}
export MODEL_B_PROFILE_RESULTS_ALIAS_NAME=${MODEL_B_PROFILE_RESULTS_ALIAS_NAME:-H100_TP1P_TP1D}
export MODEL_A_PROFILE_RESULTS_DIR=${MODEL_A_PROFILE_RESULTS_DIR:-$PROFILE_RESULTS_STAGING_ROOT/$MODEL_A_PROFILE_RESULTS_ALIAS_NAME}
export MODEL_B_PROFILE_RESULTS_DIR=${MODEL_B_PROFILE_RESULTS_DIR:-$PROFILE_RESULTS_STAGING_ROOT/$MODEL_B_PROFILE_RESULTS_ALIAS_NAME}
export LEGACY_PROFILE_RESULTS_DIR=/workspace/components/src/dynamo/planner/tests/data/profiling_results/H100_TP1P_TP1D

if [ "$MODEL_A_PROFILE_RESULTS_DIR" = "$LEGACY_PROFILE_RESULTS_DIR" ] || [ "$MODEL_B_PROFILE_RESULTS_DIR" = "$LEGACY_PROFILE_RESULTS_DIR" ]; then
  echo "Refusing to deploy with legacy MODEL_*_PROFILE_RESULTS_DIR=$LEGACY_PROFILE_RESULTS_DIR"
  echo "Unset stale shell exports first: unset MODEL_A_PROFILE_RESULTS_DIR MODEL_B_PROFILE_RESULTS_DIR"
  exit 1
fi

: "${STORAGE_CLASS_NAME:?Set STORAGE_CLASS_NAME to an RWX-capable StorageClass}"
: "${DYNAMO_IMAGE:?Set DYNAMO_IMAGE for the frontend/router/planner image}"
: "${DYNAMO_VLLM_IMAGE:?Set DYNAMO_VLLM_IMAGE for the vLLM worker image}"

export RENDERED=/tmp/global-planner-single-frontend-two-models.rendered.yaml

# 1) Render and inspect the final manifest
envsubst < global-planner-single-frontend-two-models.yaml > "$RENDERED"
test "$(rg -c '^kind: ClusterRoleBinding$' "$RENDERED")" = 1
test "$(rg -c '^kind: ConfigMap$' "$RENDERED")" = 2
test "$(rg -c '^kind: PersistentVolumeClaim$' "$RENDERED")" = 1
test "$(rg -c '^kind: DynamoGraphDeployment$' "$RENDERED")" = 7
! rg -n '\$\{[A-Z_][A-Z0-9_]*\}' "$RENDERED"
! rg -n "$LEGACY_PROFILE_RESULTS_DIR" "$RENDERED"
! rg -n 'image:\s*$|image:\s*null$' "$RENDERED"

# 2) Validate shape, then apply
# Some kubectl builds still contact the API server for discovery even with
# --dry-run=client, so this step expects a reachable cluster context.
kubectl apply --dry-run=client -f "$RENDERED"
# Optional when connected to a cluster with Dynamo CRDs installed:
# kubectl apply --dry-run=server -f "$RENDERED"
kubectl apply -f "$RENDERED"

# 3) Inspect the created resources
kubectl -n "$K8S_NAMESPACE" get configmap gp-vllm-model-a-router-config gp-vllm-model-b-router-config
kubectl -n "$K8S_NAMESPACE" get pvc hf-model-cache-vllm-two-models
kubectl -n "$K8S_NAMESPACE" get dynamographdeployment \
  gp-vllm-ctrl \
  gp-vllm-model-a-prefill-0 gp-vllm-model-a-prefill-1 gp-vllm-model-a-decode-0 \
  gp-vllm-model-b-prefill-0 gp-vllm-model-b-prefill-1 gp-vllm-model-b-decode-0

# 4) Verify the single frontend exposes both models
kubectl -n "$K8S_NAMESPACE" port-forward svc/gp-vllm-ctrl-frontend 8000:8000 &
curl -s http://127.0.0.1:8000/v1/models | head -c 4000
echo

curl -s http://127.0.0.1:8000/v1/chat/completions \
  -H "Content-Type: application/json" \
  -d "{
    \"model\": \"${MODEL_A}\",
    \"messages\": [{\"role\":\"user\",\"content\":\"Say hello from model A in one short sentence.\"}],
    \"temperature\": 0.2,
    \"max_tokens\": 64
  }" | head -c 2000
echo

curl -s http://127.0.0.1:8000/v1/chat/completions \
  -H "Content-Type: application/json" \
  -d "{
    \"model\": \"${MODEL_B}\",
    \"messages\": [{\"role\":\"user\",\"content\":\"Say hello from model B in one short sentence.\"}],
    \"temperature\": 0.2,
    \"max_tokens\": 64
  }" | head -c 2000
echo

# 5) Inspect shared-cache mounts and the pool DGDs behind the single frontend
kubectl -n "$K8S_NAMESPACE" get pods -o yaml | rg 'hf-model-cache-vllm-two-models|/home/dynamo/.cache/huggingface/hub'
kubectl -n "$K8S_NAMESPACE" get pods | rg 'gp-vllm-ctrl|gp-vllm-model-a|gp-vllm-model-b'

# 6) Delete the example
kubectl delete -f "$RENDERED"
